Animated Rings /* Este es un ejemplo genérico de CSS que genera el efecto de anillos 3D rotatorios y decrecientes. */ :root { --ring-color: #00ff44; /* Color base para los anillos */ --ring-size: 100px; } * { margin: 0; padding: 0; box-sizing: border-box; } body { display: flex; justify-content: center; align-items: center; min-height: 100vh; background: #000; overflow: hidden; /* Oculta posibles barras de desplazamiento */ } .container { position: relative; /* Aplica una perspectiva para el efecto 3D */ transform-style: preserve-3d; perspective: 800px; } .circle { position: absolute; width: var(--ring-size); height: var(--ring-size); border: 3px solid var(--ring-color); border-radius: 50%; /* Hace que el color del anillo exterior se desvanezca */ border-color: var(--ring-color) transparent transparent transparent; /* Centra el anillo en el contenedor */ top: 50%; left: 50%; transform: translate(-50%, -50%); /* Configuración de la animación */ animation: rotate 4s linear infinite; /* Utiliza la variable --i para el retraso y el tamaño */ animation-delay: calc(var(--i) * -0.25s); /* Aplicar transformaciones 3D */ transform-origin: 0% 0%; /* Punto de origen para el escalado */ transform: translate(-50%, -50%) rotateX(60deg) /* Inclina la forma para el efecto de cono */ rotateZ(calc(var(--i) * 22.5deg)) /* Separa la rotación en Z de cada círculo */ scale(calc(1 - var(--i) * 0.05)); /* Escala cada círculo para que sea más pequeño */ /* Brillo Neon */ box-shadow: 0 0 10px var(--ring-color), 0 0 20px var(--ring-color); /* Un pequeño ajuste para los anillos más externos para variar el efecto */ opacity: calc(1 - var(--i) * 0.03); } /* Define la animación de rotación */ @keyframes rotate { 0% { transform: translate(-50%, -50%) rotateX(60deg) rotateZ(0deg) scale(var(--scale)); } 100% { transform: translate(-50%, -50%) rotateX(60deg) rotateZ(360deg) /* Rota 360 grados */ scale(var(--scale)); } }
